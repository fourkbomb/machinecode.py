REGISTERS = 2
BITS_PER_BYTE = 8
CELLS = 256

INSTRUCTIONS:

0: HALT
1: ADD 1 TO R0
2: ADD -1 TO R0
3: ADD 1 TO R1
4: ADD -1 TO R1
5: ADD R1 TO R0
6: ADD -R1 TO R0
7: PRINT R0 AS UINT
8: JUMP <DATA> IF R0 != 0
9: JUMP <DATA> IF R0 == 0
10: LOAD <DATA> TO R0
11: LOAD <DATA> TO R1
12: WRITE R0 TO <DATA>
13: WRITE R1 TO <DATA>
14: SWAP R0 WITH <DATA>
15: SWAP R1 WITH <DATA>
16: BELL
17: PRINT R0 AS CHAR
